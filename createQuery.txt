create table User( usr_num varchar(10) primary key, usr_name varchar(20), password varchar(20));

create table Log( lsn int auto_increment primary key, usr_num varchar(10) references User(usr_num), start_time datetime, washer_num varchar(20));

create table Washer( washer_num varchar(20) primary key, washer_state int, last_usr_num varchar(10) references User(usr_num), location varchar(50));
